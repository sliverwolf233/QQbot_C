请将json文件修改为:
{
    "databaseencryption" : "false",
    "password" : "如果你在databaseseencryption设置为true 这里请设置密码",
    "QQRC":[
    {"AppID" :  "你在QQ开放平台的AppID",
    "clientSecret" : "你在QQ开放平台的ClientSecret"},
    "QBName" : "你在QQ开放平台备案的名字"
    {
    您如果需要同时运行多个bot 可以多添加几个配置.
    }]
}
